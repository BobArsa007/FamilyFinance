import pandas as pd import json # Load the data mama_pakde_df = pd.read_csv('Spreadsheet Manajemen Finansial.xlsx - Keuangan Keluarga.csv') hutang_df = pd.read_csv('Spreadsheet Manajemen Finansial.xlsx - Manajemen Hutang.csv') # --- PROCESS KEUANGAN KELUARGA --- # Mama's data is cols 0-5. Pakde's is 7-12. def clean_ledger(df, start_col, end_col): subset = df.iloc[1:, start_col:end_col+1].copy() subset.columns = ['Tanggal', 'Deskripsi', 'Kategori', 'Pemasukan', 'Pengeluaran', 'Saldo'] subset = subset.dropna(subset=['Tanggal', 'Deskripsi', 'Saldo'], how='all') return subset.to_dict(orient='records') mama_data = clean_ledger(mama_pakde_df, 0, 5) pakde_data = clean_ledger(mama_pakde_df, 7, 12) # Calculate totals for summary cards def get_latest_balance(data): for row in reversed(data): if pd.notnull(row.get('Saldo')): try: return float(str(row['Saldo']).replace(',', '')) except: continue return 0 mama_balance = get_latest_balance(mama_data) pakde_balance = get_latest_balance(pakde_data) # --- PROCESS MANAJEMEN HUTANG --- # PLN is cols 0-2 (header at row 2). Bank Jateng is cols 4-7. pln_raw = hutang_df.iloc[2:, 0:3].copy() pln_raw.columns = ['No', 'Bulan', 'Nominal'] pln_data = pln_raw.dropna(subset=['Nominal']).to_dict(orient='records') bank_raw = hutang_df.iloc[2:, 4:8].copy() bank_raw.columns = ['No', 'Bulan', 'Nominal', 'Status'] bank_data = bank_raw.dropna(subset=['Nominal']).to_dict(orient='records') def calc_remaining(data): total = 0 for row in data: status = str(row.get('Status', '')).upper() if status != 'LUNAS': try: total += float(str(row['Nominal']).replace(',', '')) except: continue return total total_debt = calc_remaining(pln_data) + calc_remaining(bank_data) # --- CONSTRUCT HTML --- html_content = f""" Finance Management Dashboard

Financial Management Tool

Tracking and analytics for family accounts and debts.

Saldo Mama
Rp {mama_balance:,.0f}
Saldo Pakde Fajar
Rp {pakde_balance:,.0f}
Total Sisa Hutang
Rp {total_debt:,.0f}

Log Transaksi Mama

Tanggal Deskripsi Pemasukan Pengeluaran Saldo

Log Transaksi Pakde Fajar

Tanggal Deskripsi Pemasukan Pengeluaran Saldo

Hutang PLN

NoBulanNominal

Hutang Bank Jateng

NoBulanNominalStatus
""" with open('finance_management_tool.html', 'w') as f: f.write(html_content)
